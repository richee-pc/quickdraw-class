# 앱스크립트 수집기 연동 가이드

`나만의_퀵드로우_수집기.html` 의 DB 연동은 아래 형태의 API를 기대합니다.

- `POST ?` with body:
  - `{ action: "append", classId, objectName, records: [{ studentId, vec, meta, createdAt }] }`
- `GET ?action=list&classId=...&objectName=...`
  - 응답: `{ records: [...] }`

아래 코드를 Google Apps Script(Web App)로 배포하면 바로 연결할 수 있습니다.

```javascript
const SHEET_NAME = "collector_records";

function getSheet_() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sh = ss.getSheetByName(SHEET_NAME);
  if (!sh) {
    sh = ss.insertSheet(SHEET_NAME);
    sh.appendRow(["classId", "objectName", "studentId", "createdAt", "vecJson", "metaJson"]);
  }
  return sh;
}

function doPost(e) {
  try {
    const body = JSON.parse(e.postData.contents || "{}");
    if (body.action !== "append") {
      return json_({ ok: false, error: "unsupported action" });
    }
    const classId = body.classId || "class-a";
    const objectName = body.objectName || "object";
    const records = body.records || [];
    const sh = getSheet_();

    records.forEach((r) => {
      sh.appendRow([
        classId,
        objectName,
        r.studentId || "anonymous",
        r.createdAt || new Date().toISOString(),
        JSON.stringify(r.vec || []),
        JSON.stringify(r.meta || {}),
      ]);
    });

    return json_({ ok: true, saved: records.length });
  } catch (err) {
    return json_({ ok: false, error: String(err) });
  }
}

function doGet(e) {
  try {
    const action = (e.parameter.action || "").toLowerCase();
    if (action !== "list") return json_({ ok: false, error: "unsupported action" });

    const classId = e.parameter.classId || "class-a";
    const objectName = e.parameter.objectName || "object";
    const sh = getSheet_();
    const values = sh.getDataRange().getValues();
    const out = [];
    for (let i = 1; i < values.length; i++) {
      const [cId, obj, studentId, createdAt, vecJson, metaJson] = values[i];
      if (cId !== classId || obj !== objectName) continue;
      out.push({
        classId: cId,
        objectName: obj,
        studentId: studentId,
        createdAt: createdAt,
        vec: JSON.parse(vecJson || "[]"),
        meta: JSON.parse(metaJson || "{}"),
      });
    }
    return json_({ ok: true, records: out });
  } catch (err) {
    return json_({ ok: false, error: String(err) });
  }
}

function json_(obj) {
  return ContentService.createTextOutput(JSON.stringify(obj)).setMimeType(ContentService.MimeType.JSON);
}
```

## 배포 순서

1. 구글 스프레드시트 새로 만들기
2. `확장 프로그램 → Apps Script` 열기
3. 위 코드 붙여넣고 저장
4. `배포 → 새 배포 → 웹 앱`
   - 실행 사용자: 본인
   - 접근 권한: 링크가 있는 모든 사용자
5. `.../exec` URL을 복사해서 수집기의 `데이터베이스 API URL`에 입력
6. `수업 코드(classId)`를 학급별로 다르게 설정해서 사용
